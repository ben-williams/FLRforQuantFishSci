% gets rid of navigation symbols
\setbeamertemplate{navigation symbols}{}

% colors
\definecolor{drk-blue}{RGB}{0, 51, 102} % blue text
\definecolor{jrc-blue}{RGB}{0, 184, 252} % blue header

\setbeamercolor{shadedhead}{bg=jrc-blue}

% footer image
\pgfdeclareimage[height=5.5mm]{footimage}{tex/footer}

% header image
\pgfdeclareimage[height=1.5cm]{headimage}{tex/header}

%% footline
%% left: date as in title
\newcommand{\footlinetext}{\insertdate}
\setbeamertemplate{footline}{
\begin{minipage}{58mm}
 \color{drk-blue} \hspace{7.5mm} \footlinetext
\end{minipage}
\hspace{0.1mm}
%% center: jrc logo
\begin{minipage}{8.5mm}
 \pgfuseimage{footimage}
\end{minipage}
\hspace{0.1mm}
%% right: page number
\begin{minipage}{50mm}
 \color{drk-blue} 
 \flushright \insertframenumber
\end{minipage}
}

%% headline
\setbeamertemplate{headline}{
\leavevmode\hbox{%
% uniform
\begin{beamercolorbox}[wd=\textwidth,ht=0.13\paperheight,dp=0ex]{shadedhead}%
	% EC logo
	\centering
  \vspace{-1cm}
	\pgfuseimage{headimage}
\end{beamercolorbox}%
}}


% bullet points
\setbeamertemplate{itemize items}[circle] % if you want a ball
\setbeamertemplate{itemize subitem}[circle] % if you wnat a circle
\setbeamertemplate{itemize subsubitem}[circle] % if you want a triangle
